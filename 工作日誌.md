點數店家報表
店家都沒有使用點數
目前有點數的店家 每個月的使用狀況
李襄
``` mysql

SET @s = '2020-05-01 00:00:00';
SET @e = DATE_ADD(@s, INTERVAL 100 MONTH);

SELECT b.company_name '公司名稱',b.corporate_brand '公司品牌',b.tax_id 'ID',a.point_name'點數類型',case when a.points>0 then '贈點' ELSE '扣點' END '用途'
		,a.user_no'會員編號',a.user_name'姓名',a.user_link'手機號碼',a.points'點數',a.source'備註',a.created_at'交易日期'
FROM sunnyds.yxds_users_points_record a 
JOIN sunnyds.yxds_store b ON a.store_id=b.id
WHERE a.store_id>0
AND a.branch_id>0
AND a.operation_type IN (2,3)
AND a.created_at >= @s 
AND a.created_at < @e
ORDER BY a.id desc
;
```
